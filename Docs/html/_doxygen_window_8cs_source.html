309  void GenerateGUI()
310  {
311  if(DoxyFileExists)
312  {
313  GUILayout.Space (10);
314  if(!DocsGenerated)
315  GUI.enabled = false;
316  if(GUILayout.Button ("Browse Documentation", GUILayout.Height(40)))
317  Application.OpenURL("File://"+Config.DocDirectory+"/html/index.html");
318  GUI.enabled = true;
319 
320  if(DoxygenOutput == null)
321  {
322  if(GUILayout.Button ("Run Doxygen", GUILayout.Height(40)))
323  {
324  DocsGenerated = false;
325  RunDoxygen();
326  }
327 
328  if(DocsGenerated && DoxygenLog != null)
329  {
330  if(GUILayout.Button( "View Doxygen Log",EditorStyles.toolbarDropDown))
331  ViewLog = !ViewLog;
332  if(ViewLog)
333  {
334  scroll = EditorGUILayout.BeginScrollView(scroll, GUILayout.ExpandHeight(true));
335  foreach(string logitem in DoxygenLog)
336  {
337  EditorGUILayout.SelectableLabel(logitem,EditorStyles.miniLabel,GUILayout.ExpandWidth(true));
338  }
339  EditorGUILayout.EndScrollView();
340  }
341  }
342  }
343  else
344  {
345  if(DoxygenOutput.isStarted() && !DoxygenOutput.isFinished())
346  {
347  string currentline = DoxygenOutput.ReadLine();
348  DoxyoutputProgress = DoxyoutputProgress + 0.1f;
349  if(DoxyoutputProgress >= 0.9f)
350  DoxyoutputProgress = 0.75f;
351  Rect r = EditorGUILayout.BeginVertical();
352  EditorGUI.ProgressBar(r, DoxyoutputProgress,currentline );
353  GUILayout.Space(40);
354  EditorGUILayout.EndVertical();
355  }
356  if(DoxygenOutput.isFinished())
357  {
358  if (Event.current.type == EventType.Repaint)
359  {
360  /*
361  If you look at what SetTheme is doing, I know, it seems a little scary to be
362  calling file moving operations from inside a an OnGUI call like this. And
363  maybe it would be a better choice to call SetTheme and update these other vars
364  from inside of the OnDoxygenFinished callback. But since the callback is static
365  that would require a little bit of messy singleton instance checking to make sure
366  the call back was calling into the right functions. I did try to do it that way
367  but for some reason the file operations failed every time. I'm not sure why.
368  This is what I was getting from the debugger:
369 
370  Error in file: C:/BuildAgent/work/842f9551727e852/Runtime/Mono/MonoManager.cpp at line 2212
371  UnityEditor.FileUtil:DeleteFileOrDirectory(String)
372  UnityEditor.FileUtil:ReplaceFile(String, String) (at C:\BuildAgent\work\842f9557127e852\Editor\MonoGenerated\Editor\FileUtil.cs:42)
373 
374  Doing them here seems to work every time and the Repaint event check ensures that they will only be done once.
375  */
376  SetTheme(SelectedTheme);
377  DoxygenLog = DoxygenOutput.ReadFullLog();
378  DoxyoutputProgress = -1.0f;
379  DoxygenOutput = null;
380  DocsGenerated = true;
381  EditorPrefs.SetBool(UnityProjectID+"DocsGenerated",DocsGenerated);
382  }
383  }
384  }
385  }
386  else
387  {
388  GUIStyle ErrorLabel = new GUIStyle(EditorStyles.largeLabel);
389  ErrorLabel.alignment = TextAnchor.MiddleCenter;
390  GUILayout.Space(20);
391  GUI.contentColor = Color.red;
392  GUILayout.Label("You must set the path to your Doxygen install and \nbuild a new Doxyfile before you can generate documentation",ErrorLabel);
393  }

703  public static int Run(Action<string> output, TextReader input, string exe, params string[] args)
704  {
705  if (String.IsNullOrEmpty(exe))
706  throw new FileNotFoundException();
707  if (output == null)
708  throw new ArgumentNullException("output");
709 
710  ProcessStartInfo psi = new ProcessStartInfo();
711  psi.UseShellExecute = false;
712  psi.RedirectStandardError = true;
713  psi.RedirectStandardOutput = true;
714  psi.RedirectStandardInput = true;
715  psi.WindowStyle = ProcessWindowStyle.Hidden;
716  psi.CreateNoWindow = true;
717  psi.ErrorDialog = false;
718  psi.WorkingDirectory = WorkingFolder;
719  psi.FileName = FindExePath(exe);
720  psi.Arguments = EscapeArguments(args);
721 
722  using (Process process = Process.Start(psi))
723  using (ManualResetEvent mreOut = new ManualResetEvent(false),
724  mreErr = new ManualResetEvent(false))
725  {
726  process.OutputDataReceived += (o, e) => { if (e.Data == null) mreOut.Set(); else output(e.Data); };
727  process.BeginOutputReadLine();
728  process.ErrorDataReceived += (o, e) => { if (e.Data == null) mreErr.Set(); else output(e.Data); };
729  process.BeginErrorReadLine();
730 
731  string line;
732  while (input != null && null != (line = input.ReadLine()))
733  process.StandardInput.WriteLine(line);
734 
735  process.StandardInput.Close();
736  process.WaitForExit();
737 
738  mreOut.WaitOne();
739  mreErr.WaitOne();
740  return process.ExitCode;
741  }
742  }

752  public static string EscapeArguments(params string[] args)
753  {
754  StringBuilder arguments = new StringBuilder();
755  Regex invalidChar = new Regex("[\x00\x0a\x0d]");// these can not be escaped
756  Regex needsQuotes = new Regex(@"\s|""");// contains whitespace or two quote characters
757  Regex escapeQuote = new Regex(@"(\\*)(""|$)");// one or more '\' followed with a quote or end of string
758  for (int carg = 0; args != null && carg < args.Length; carg++)
759  {
760  if (args[carg] == null)
761  {
762  throw new ArgumentNullException("args[" + carg + "]");
763  }
764  if (invalidChar.IsMatch(args[carg]))
765  {
766  throw new ArgumentOutOfRangeException("args[" + carg + "]");
767  }
768  if (args[carg] == String.Empty)
769  {
770  arguments.Append("\"\"");
771  }
772  else if (!needsQuotes.IsMatch(args[carg]))
773  {
774  arguments.Append(args[carg]);
775  }
776  else
777  {
778  arguments.Append('"');
779  arguments.Append(escapeQuote.Replace(args[carg], m =>
780  m.Groups[1].Value + m.Groups[1].Value +
781  (m.Groups[2].Value == "\"" ? "\\\"" : "")
782  ));
783  arguments.Append('"');
784  }
785  if (carg + 1 < args.Length)
786  arguments.Append(' ');
787  }
788  return arguments.ToString();
789  }

799  public static string FindExePath(string exe)
800  {
801  exe = Environment.ExpandEnvironmentVariables(exe);
802  if (!File.Exists(exe))
803  {
804  if (Path.GetDirectoryName(exe) == String.Empty)
805  {
806  foreach (string test in (Environment.GetEnvironmentVariable("PATH") ?? "").Split(';'))
807  {
808  string path = test.Trim();
809  if (!String.IsNullOrEmpty(path) && File.Exists(path = Path.Combine(path, exe)))
810  return Path.GetFullPath(path);
811  }
812  }
813  throw new FileNotFoundException(new FileNotFoundException().Message, exe);
814  }
815  return Path.GetFullPath(exe);
816  }
